class Solution(object):
    def isValid(self, s):
        fermants = {
            ')': '(',
            ']': '[',
            '}': '{'
        }

        pile = []

        for char in s:
            if char not in fermants:
                pile.append(char)
            else:
                if not pile:
                    return False

                if pile.pop() != fermants[char]:
                    return False

        return not pile
